CREATE TABLE IF NOT EXISTS fictions (
    id INT,
    title VARCHAR(100), 
    description VARCHAR(3000),
    rating DECIMAL(4, 2),
    chapters INT,
    retrieved_time DATETIME,
    url VARCHAR(200),
    followers INT, 
    pages INT,
    views INT,
    CONSTRAINT PK_Fictions PRIMARY KEY (id, retrieved_time)
);

CREATE TABLE IF NOT EXISTS rising_stars (
    id INT, 
    time VARCHAR(30),
    category VARCHAR(20),
    placement INT,
    CONSTRAINT PK_Rising_stars PRIMARY KEY (id, time)
);
